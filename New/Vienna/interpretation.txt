  BLOCK macros:
    BLOCK macro entity_battery_alert ($entity $alertLevel):
      COMMENT # Source: entity_battery_alert.def
      STATEMENT declare entity binary_sensor.infrastructural:$entity:battery_alert;
      BLOCK if "$alertLevel" != "" then
        STATEMENT define satisfies sensor.infrastructural:$entity:battery_level "($ | int(0)) < $alertLevel";
        STATEMENT available always;
      END BLOCK
    END BLOCK
    BLOCK macro entity_battery_from_device ($entity $sourceEntity $alertLevel):
      COMMENT # Source: entity_battery_from_device.def
      STATEMENT declare entity sensor.infrastructural:$entity:battery_level;

      STATEMENT entity battery_alert $entity $alertLevel;
    END BLOCK
    BLOCK macro entity_battery_level ($entity $alertLevel):
      COMMENT # Source: entity_battery_level.def
      STATEMENT entity battery_alert $entity $alertLevel;
      STATEMENT declare entity sensor.infrastructural:$entity:battery_level;
    END BLOCK
    BLOCK macro entity_battery_level_device ($entity $alertLevel):
      COMMENT # Source: entity_battery_level_device.def

      STATEMENT providing node $entity;
    END BLOCK
    BLOCK macro entity_light_device ($sphereOrName $name):
      COMMENT # Source: entity_light_device.def
      BLOCK if "$name" == "" then
        STATEMENT lightSphere="social";

      END BLOCK
      BLOCK else
        STATEMENT lightSphere="$sphereOrName";
        STATEMENT lightName="$name";
      END BLOCK
      STATEMENT declare entity light.$lightSphere:$lightName;
      STATEMENT providing node $lightName/light;
    END BLOCK
    BLOCK macro entity_media_player ($entity $option $value):
      COMMENT # Source: entity_media_player.def
      STATEMENT thisEntity="$entity";
      BLOCK if "$option" == "no_collect" then
        STATEMENT noCollect="no_collect";
        STATEMENT shift;
      END BLOCK
      BLOCK else
        STATEMENT noCollect="";
      END BLOCK
      BLOCK if "$option" == "no_play" then
        STATEMENT enabler="";
        STATEMENT delayOffety="";
        STATEMENT inputety="$value";
      END BLOCK
      BLOCK else
        STATEMENT enabler="$option";
        STATEMENT delayOffety="$value";
        STATEMENT inputety="";
      END BLOCK
      STATEMENT declare entity media_player.social:$thisEntity $noCollect;
      STATEMENT providing node $thisEntity $enabler $delayOffety;
      STATEMENT mediaPlayerSpecification=$CurrentEntitySpecification;
      STATEMENT declare entity switch.social:$thisEntity:media $noCollect;
      STATEMENT icon $MediaSwitchIcon;
      STATEMENT define media_switch $mediaPlayerSpecification "$inputety";
    END BLOCK
    BLOCK macro entity_power_switch ($sphere $entity $deviceNode $powerThreshold):

      STATEMENT declare entity sensor.$sphere:$entity:power;
      STATEMENT declare entity binary_sensor.$sphere:$entity:consumes;
      STATEMENT icon $ConsumesIcon;

    BLOCK macro entity_zigbee_group ($domain $sphere $location $entities...):

        END BLOCK
      END BLOCK
