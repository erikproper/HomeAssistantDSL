  BLOCK creation macro battery_alert { $alertLevel int op }:
    COMMENT # Source: entity_battery_alert.def
    STATEMENT if $alertLevel is provided then
    BLOCK entity binary_sensor.infrastructural:$entity:battery_alert with:
      STATEMENT definition as condition sensor.infrastructural:$entity:battery_level "($ | int(0)) < $alertLevel";
      STATEMENT available always;
    END BLOCK
    STATEMENT else
    STATEMENT entity binary_sensor.infrastructural:$entity:battery_alert;
  END BLOCK
  BLOCK creation macro battery_from_device { $sourceEntity entityReference, $alertLevel int op }:
    COMMENT # Source: entity_battery_from_device.def
    STATEMENT entity sensor.infrastructural:$entity:battery_level with
    STATEMENT definition as value $sourceEntity!battery_level;
    STATEMENT entity battery_alert $entity $alertLevel;
  END BLOCK
  BLOCK creation macro battery_level { $alertLevel int op }:
    COMMENT # Source: entity_battery_level.def
    STATEMENT entity battery_alert $entity $alertLevel;
    STATEMENT entity sensor.infrastructural:$entity:battery_level;
  END BLOCK
  BLOCK creation macro battery_level_device { $alertLevel int op }:

    STATEMENT entity battery_level $entity $alertLevel with providing node $entity;
  END BLOCK
  BLOCK creation macro light_device:

  END BLOCK
  BLOCK creation macro media_player { $enabler string op, $offDelay string op, $noPlayInput string op, $no_collect option}:
    BLOCK mediaPlayer = entity media_player.social:$entity with:
      STATEMENT maybe apply $no_collect;
      STATEMENT if $enabler is provided then
      STATEMENT providing node $entity $enabler $offDelay;
    END BLOCK
    BLOCK entity switch.social:mediaPlayer:media with:
      STATEMENT if $noPlayInput is provided then
      STATEMENT definition as media_switch $mediaPlayer $input;
      STATEMENT maybe apply $no_collect;
      STATEMENT icon $MediaSwitchIcon;
    END BLOCK
  END BLOCK
  BLOCK creation macro power_switch { $node string, $threshold int }:
    COMMENT # Source: entity_power_switch.def
    STATEMENT entity sensor.$sphere:$entity:power;
    BLOCK entity binary_sensor.$sphere:$entity:consumes with:
      STATEMENT definition as condition sensor.$sphere:$entity:power "($ | int(0)) > $threshold";
      STATEMENT if $icon is provided then
      STATEMENT icon $icon;
      STATEMENT else
      STATEMENT icon $ConsumesIcon;
    END BLOCK
    STATEMENT entity switch.$sphere::$entity with providing node $entity/$node;
  END BLOCK
  BLOCK creation macro switch_device { $name string op }:
    STATEMENT entity switch.$sphere:$entity with providing node $entity/switch;
  END BLOCK
  BLOCK creation macro thermostat:
    COMMENT # Source: entity_thermostat.def
    STATEMENT entity climate.physical:$entity with providing node $entity;
    BLOCK entity sensor.physical:$entity:temperature with:
      STATEMENT definition as value climate.physical:$entity!current_temperature;
    END BLOCK
  END BLOCK
  BLOCK creation macro zigbee_group { $elements set of entityReference }:
    COMMENT # Source: entity_zigbee_group.def
    COMMENT # $1 = Domain, $2 = Sphere, $3 = Location, $4 ... $n = Entity names
    STATEMENT create <$type>_device $sphere:$entity;
    BLOCK space physical:$entity with:
      STATEMENT no_collect
      STATEMENT for $element in $elements do
      STATEMENT create <$type>_device physical:$e;
    END BLOCK
  END BLOCK
  BLOCK creation macro zwave_node:
    COMMENT # Source: entity_zwave_node.def
    STATEMENT entity sensor.infrastructural:$entity/status;

  END BLOCK
  BLOCK creation macro windy { $sensor entityReference, $delay_on string op, $delay_off string op }:
    COMMENT # Windy condition: compares wind speed threshold against a sensor value.
    COMMENT # Multi-parameter condition syntax: $1 is threshold, $2 is sensor value.
    BLOCK $windyThreshold = entity input_number.social:/:windy_threshold with:
      STATEMENT defined as input_number;
      STATEMENT minimum $WindSpeedMin;
      STATEMENT maximum $WindSpeedMax;
      STATEMENT icon $WindSpeedIcon;
      STATEMENT units $WindSpeedUnit;
    END BLOCK
    BLOCK entity binary_sensor.social::windy with:
      STATEMENT definition as condition $windyThreshold $sensor "($1 | int) > ($2 | int)";
      STATEMENT device_class $WindyDeviceClass;
      STATEMENT maybe use $delay_on;
      STATEMENT maybe use $delay_off;
    END BLOCK
  END BLOCK
  BLOCK creation macro sunny { $sensor entityReference, $delay_on string op, $delay_off string op }:
    COMMENT # Sunny condition: compares illuminance threshold against a sensor value.
    COMMENT # Multi-parameter condition syntax: $1 is threshold, $2 is sensor value.
    BLOCK $sunnyThreshold = entity input_number.social:/:sunny_threshold with:
      STATEMENT defined as input_number;
      STATEMENT minimum $IlluminanceMin;
      STATEMENT maximum $IlluminanceMax;
      STATEMENT icon $IlluminanceIcon;
      STATEMENT units $IlluminanceUnit;
    END BLOCK
    BLOCK entity binary_sensor.social::sunny with:
      STATEMENT definition as condition $sunnyThreshold $sensor "($1 | int) < ($2 | int)";
      STATEMENT device_class $SunnyDeviceClass;
      STATEMENT maybe use $delay_on;
      STATEMENT maybe use $delay_off;
    END BLOCK
  END BLOCK
